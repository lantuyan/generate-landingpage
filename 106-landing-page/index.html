Connect in Minutes,
Not Months

Single API endpoint. Comprehensive documentation. Production-ready in one sprint.

integration.js 
// Initialize RouteForge client
import { RouteForge } from '@routeforge/sdk';

const rf = new RouteForge({
  apiKey: process.env.ROUTEFORGE_KEY,
  environment: 'production'
});

// Orchestrate a shipment
const shipment = await rf.orchestrate({
  origin: { zip: '10001', country: 'US' },
  destination: { zip: '90210', country: 'US' },
  package: { weight: 2.5, dimensions: [12, 8, 6] },
  priority: 'optimal' // cost vs speed
});

// Response with optimal carrier selected
console.log(shipment);
// → { carrier: 'UPS', rate: 12.47, eta: '2024-01-15' }
